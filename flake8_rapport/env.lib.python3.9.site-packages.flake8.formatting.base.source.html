117 def show_statistics(self, statistics: "Statistics") -> None:
118 """Format and print the statistics."""
119 for error_code in statistics.error_codes():
120 stats_for_error_code = statistics.statistics_for(error_code)
121 statistic = next(stats_for_error_code)
122 count = statistic.count
123 count += sum(stat.count for stat in stats_for_error_code)
124 self._write(f"{count:<5} {error_code} {statistic.message}")

126 def show_benchmarks(self, benchmarks: List[Tuple[str, float]]) -> None:
127 """Format and print the benchmarks."""
128 # NOTE(sigmavirus24): The format strings are a little confusing, even
129 # to me, so here's a quick explanation:
130 # We specify the named value first followed by a ':' to indicate we're
131 # formatting the value.
132 # Next we use '<' to indicate we want the value left aligned.
133 # Then '10' is the width of the area.
134 # For floats, finally, we only want only want at most 3 digits after
135 # the decimal point to be displayed. This is the precision and it
136 # can not be specified for integers which is why we need two separate
137 # format strings.
138 float_format = "{value:<10.3} {statistic}".format
139 int_format = "{value:<10} {statistic}".format
140 for statistic, value in benchmarks:
141 if isinstance(value, int):
142 benchmark = int_format(statistic=statistic, value=value)
143 else:
144 benchmark = float_format(statistic=statistic, value=value)
145 self._write(benchmark)

147 def show_source(self, error: "Violation") -> Optional[str]:
148 """Show the physical line generating the error.
149  
150 This also adds an indicator for the particular part of the line that
151 is reported as generating the problem.
152  
153 :param error:
154 This will be an instance of
155 :class:`~flake8.violation.Violation`.
156 :returns:
157 The formatted error string if the user wants to show the source.
158 If the user does not want to show the source, this will return
159 ``None``.
160 """
161 if not self.options.show_source or error.physical_line is None:
162 return ""
163  
164 # Because column numbers are 1-indexed, we need to remove one to get
165 # the proper number of space characters.
166 indent = "".join(
167 c if c.isspace() else " "
168 for c in error.physical_line[: error.column_number - 1]
169 )
170 # Physical lines have a newline at the end, no need to add an extra
171 # one
172 return f"{error.physical_line}{indent}^"
