18 """Generate filenames from an argument.
19  
20 :param arg:
21 Parameter from the command-line.
22 :param predicate:
23 Predicate to use to filter out filenames. If the predicate
24 returns ``True`` we will exclude the filename, otherwise we
25 will yield it. By default, we include every filename
26 generated.
27 :returns:
28 Generator of paths
29 """
30 if predicate(arg):
31 return
32  
33 if os.path.isdir(arg):
34 for root, sub_directories, files in os.walk(arg):
35 # NOTE(sigmavirus24): os.walk() will skip a directory if you
36 # remove it from the list of sub-directories.
37 for directory in tuple(sub_directories):
38 joined = os.path.join(root, directory)
39 if predicate(joined):
40 sub_directories.remove(directory)
41  
42 for filename in files:
43 joined = os.path.join(root, filename)
44 if not predicate(joined):
45 yield joined
46 else:
47 yield arg
